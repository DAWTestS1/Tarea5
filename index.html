Control de Flujo

if...else


let numero = 5;
if (numero > 0) {
    console.log('El número es positivo');
} else {
    console.log('El número es negativo o cero');

switch


let dia = 2;
switch (dia) {
    case 1:
        console.log('Lunes');
        break;
    case 2:
        console.log('Martes');
        break;
    default:
        console.log('Otro día');
}

Bucles e Iteración

for


for (let i = 0; i < 5; i++) {
    console.log(i);
}

while


let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

do...while


let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

Funciones

Definir funciones


function sumar(a, b) {
    return a + b;
}

Llamar funciones


console.log(sumar(3, 4)); // 7

Objetos y Manejo de Errores

try/catch/throw


try {
    throw new Error('¡Algo salió mal!');
} catch (e) {
    console.error(e.message);
}

Objetos Error


const error = new Error('Este es un error');
console.log(error.message);

Operadores

Operadores Aritméticos


let a = 10;
let b = 5;
console.log(a + b); // Suma
console.log(a - b); // Resta
console.log(a * b); // Multiplicación
console.log(a / b); // División
console.log(a % b); // Módulo

Operadores Lógicos


let x = true;
let y = false;
console.log(x && y); // AND
console.log(x || y); // OR
console.log(!x); // NOT

Números y Fechas

Uso del objeto Number


let numero = Number('42');
console.log(numero); // 42

Uso del objeto Math


console.log(Math.sqrt(16)); // 4
console.log(Math.random()); // Número aleatorio entre 0 y 1

Uso del objeto Date


let fecha = new Date();
console.log(fecha); // Fecha y hora actuales

Colecciones Indexadas

Arreglos


let arreglo = [1, 2, 3, 4, 5];
console.log(arreglo[0]); // 1

arregloTipado[0] = 10;
